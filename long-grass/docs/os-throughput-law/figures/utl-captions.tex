\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_01_universal_law.png}
    \caption{\textbf{Universal Transport Law: calibration against an independent Monte Carlo estimator.}
    (\textbf{A})~Predicted versus measured $TP^{-1}$ on linear axes for $20$ randomly drawn synthetic kernels with $N \in \{2,3,4,5\}$ decision classes; entries of $\tau$ drawn uniformly from $[0.5, 5.0]$ and entries of $g$ symmetrised from $[0.1, 1.0]$. Steelblue circles are the Monte Carlo estimator at $n_\mathrm{dec}=2 \times 10^{4}$ uniform samples; the crimson dashed line is $y=x$. The estimator recovers $TP^{-1}=N^{-2}\sum_{i,j}\tau^{(ij)} g^{(ij)}$ with maximum relative error $7.12 \times 10^{-3}$ and mean relative error $2.64 \times 10^{-3}$. This is the empirical content of Theorem~\ref{thm:utl} (Universal Transport Law).
    (\textbf{B})~Per-trial residual: relative error $|TP^{-1}_\mathrm{predicted} - TP^{-1}_\mathrm{measured}| / TP^{-1}_\mathrm{predicted}$ on the linear $y$-axis versus trial index $1$ to $20$ on the linear $x$-axis. The $0.10$ tolerance threshold (crimson dashed reference) is more than an order of magnitude above every measured residual; zero violations.
    (\textbf{C})~Histogram of relative errors across all $20$ trials in $12$ equal-width bins (lightblue bars, navy edges); the rightmost bin contains the maximum value $7.12 \times 10^{-3}$, far below the acceptance tolerance.
    (\textbf{D})~Three-dimensional surface of the integrand $\tau \cdot g$ over $\tau \in [0.5, 5.0]$, $g \in [0.1, 1.0]$, blues colourmap, viewing angle elevation $22^{\circ}$ azimuth $-58^{\circ}$. The volume under this surface, divided by $N^{2}$, is the universal-law prediction; the Monte Carlo estimator in panel~A independently estimates the same quantity.}
    \label{fig:utl-panel-1}
\end{figure*}

\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_02_queueing.png}
    \caption{\textbf{Queueing-theoretic specialisations: M/M/1, Jackson, and cascade.}
    (\textbf{A})~M/M/1 specialisation: $TP^{-1}$ on the linear $y$-axis versus utilisation $\rho \in [0.05, 0.95]$ on the linear $x$-axis, $27$ $(\rho, \mu)$ pairs. Navy line is the closed-form $TP^{-1}_{M/M/1} = \rho/\mu$ and amber circles are the universal-law evaluation; maximum absolute residual $2.78 \times 10^{-17}$ (machine precision). This is the empirical content of Proposition~\ref{prop:mm1-spec}: the universal law reduces to M/M/1 under the single-class restriction.
    (\textbf{B})~Jackson-network independence: composite $TP^{-1}$ via the universal law on the linear $x$-axis versus the Jackson product-form formula on the linear $y$-axis, $15$ random networks of $N \in \{2, \ldots, 6\}$ stations with diagonal coupling. Steelblue circles fall exactly on the crimson dashed identity line; maximum absolute residual $2.22 \times 10^{-16}$ (machine precision). This is the empirical content of Proposition~\ref{prop:jackson-spec}.
    (\textbf{C})~Cascade serialisation: $TP^{-1}$ on the linear $y$-axis versus cascade depth $k \in \{1, \ldots, 15\}$ on the linear $x$-axis. Navy circles are the cascade-sum formula $TP^{-1} = \sum_i \tau_i g_i$ and amber crosses are the universal-law evaluation under banded coupling; the two curves coincide. This is the empirical content of Proposition~\ref{prop:cascade-spec}: the universal law reduces to pipeline serialisation when coupling is concentrated on neighbours.
    (\textbf{D})~Three-dimensional surface of $\log_{10}(W)$ where $W = \rho/(\mu(1-\rho))$ is the M/M/1 mean wait, over $\rho \in [0.05, 0.95]$, $\mu \in [0.5, 5.0]$, blues colourmap, viewing angle elevation $24^{\circ}$ azimuth $-50^{\circ}$. The steep ridge along $\rho \to 1$ visualises the saturation regime predicted by Theorem~\ref{thm:utl} when $g$ has rank one.}
    \label{fig:utl-panel-2}
\end{figure*}

\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_03_regimes.png}
    \caption{\textbf{Five operating regimes of the load coordinate $R$.}
    (\textbf{A})~Partition of $R \in [0,1]$ on the linear $x$-axis into five regimes: \textsc{turbulent} ($R < 0.3$, palest blue), \textsc{aperture-dominated} ($0.3 \leq R < 0.5$, light blue), \textsc{cascade} ($0.5 \leq R < 0.8$, mid blue), \textsc{coherent} ($0.8 \leq R < 0.95$, dark blue), and \textsc{phase-locked} ($R \geq 0.95$, near-black). Boundary values $\{0.3, 0.5, 0.8, 0.95\}$ are marked by dotted vertical lines. This is the empirical content of Theorem~\ref{thm:five-regimes}.
    (\textbf{B})~Regime occupancy: bar chart of count of $R$ samples in each regime over $1001$ uniformly spaced $R$-values: turbulent $300$, aperture $200$, cascade $300$, coherent $150$, phase-locked $51$. The widths follow the analytic regime widths $0.3 : 0.2 : 0.3 : 0.15 : 0.05$.
    (\textbf{C})~Bifurcation curve $R(K)$ across $50$ coupling values $K \in [0.1, 5.0]$. Navy line is the closed-form $R^* = \sqrt{1 - K_c/K}$ for $K > K_c$, $R^* = 0$ otherwise; amber circles are simulation; maximum absolute residual $0.0$. The four regime thresholds appear as horizontal dotted lines. This is the empirical content of Theorem~\ref{thm:bifurcation}: $R$ undergoes a pitchfork bifurcation at $K_c$.
    (\textbf{D})~Three-dimensional regime cone: phase coherence on the vertical axis as a surface over $R \in [0.05, 0.99]$ and external perturbation $\epsilon \in [0, 0.4]$, blues colourmap, viewing angle elevation $22^{\circ}$ azimuth $-60^{\circ}$. The cone collapses toward zero coherence as $\epsilon$ grows, recovering the perturbed-oscillator response that motivates the regime classification.}
    \label{fig:utl-panel-3}
\end{figure*}

\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_04_critical.png}
    \caption{\textbf{Critical slowing as a load indicator.}
    (\textbf{A})~Relaxation time $\tau_\mathrm{relax}$ on the log $y$-axis versus distance $|R - R_b|$ from the nearest regime boundary $R_b$ on the log $x$-axis, $24$ measurements spanning $|R-R_b| \in [10^{-3}, 1]$. Navy circles are simulated $\tau_\mathrm{relax}$, amber crosses are the predicted $\propto |R-R_b|^{-1}$, and the crimson dashed reference traces the closed-form scaling. Both follow the law over three decades. This is the empirical content of Theorem~\ref{thm:critical-slowing}: $\tau_\mathrm{relax}$ diverges as the operating point approaches a regime boundary, with exponent $-1$.
    (\textbf{B})~Load indicator: estimated distance-to-boundary $\hat d$ on the linear $y$-axis versus true distance $d$ on the linear $x$-axis, $50$ trials. Navy markers (correct classification, $50/50$ trials) lie exactly on the crimson identity line; the indicator achieves $100\%$ classification accuracy, the empirical content of Algorithm~\ref{alg:load} together with Theorem~\ref{thm:critical-slowing}.
    (\textbf{C})~Per-trial relative error $|\hat d - d|/d$ on the linear $y$-axis versus trial index $1$ to $50$ on the linear $x$-axis. The mean residual is shown in the title; all trials lie within numerical tolerance.
    (\textbf{D})~Three-dimensional surface of $\log_{10}(\tau_\mathrm{relax})$ over $R \in [0.1, 0.95]$, perturbation $\epsilon \in [0.01, 0.3]$, blues colourmap, viewing angle elevation $22^{\circ}$ azimuth $-50^{\circ}$. The structural divergence near regime boundaries is what the load indicator exploits.}
    \label{fig:utl-panel-4}
\end{figure*}

\begin{figure*}[!htbp]
    \centering
    \includegraphics[width=\textwidth]{figures/panel_05_federation.png}
    \caption{\textbf{Cache extinction and federation composition.}
    (\textbf{A})~Cache extinction: $TP^{-1}$ after caching on the linear $y$-axis versus uncached $TP^{-1}$ on the linear $x$-axis, $20$ trials. Navy circles are the cached evaluation, crimson dashed reference is the no-cache baseline. The structural drop --- mean speedup $1.16\times$, range $[1.07\times, 1.30\times]$ --- corresponds to extinction of decision classes whose lookups become constant-time. This is the empirical content of Theorem~\ref{thm:cache-extinction}.
    (\textbf{B})~Federation composition: composite $TP^{-1}_\mathrm{fed}$ on the linear $y$-axis versus number of federated kernels $n \in \{1, \ldots, 20\}$ on the linear $x$-axis, $40$ random federations. Navy circles are simulation; the curve saturates at $\Sigma$ from below. Maximum absolute residual against the closed-form $1 - TP^{-1}_\mathrm{fed}/\Sigma = \prod_i (1 - TP^{-1}_i/\Sigma)$ is $0.0$. This is the empirical content of Theorem~\ref{thm:federation}.
    (\textbf{C})~Diminishing returns: marginal $\Delta TP^{-1}$ from adding the $n$-th kernel on the linear $y$-axis versus $n \in \{1, \ldots, 20\}$ on the linear $x$-axis, lightblue bars with navy edges. Bars decrease monotonically; final asymptote gap $\Sigma - TP^{-1}_\mathrm{fed} = 7.98 \times 10^{-2}$ at $n=20$ confirms saturation. This is the empirical content of Corollary~\ref{cor:saturation}.
    (\textbf{D})~Three-dimensional composition surface: $TP^{-1}_\mathrm{fed}$ over $n \in \{1, \ldots, 20\}$ and per-kernel $TP^{-1} \in [5, 95]$ at fixed $\Sigma = 100$, blues colourmap, viewing angle elevation $22^{\circ}$ azimuth $-58^{\circ}$. The upper plateau visualises the asymptotic ceiling that no finite federation can exceed.}
    \label{fig:utl-panel-5}
\end{figure*}
